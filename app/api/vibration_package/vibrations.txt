DROP TABLE IF EXISTS vibrations;
CREATE TABLE vibration (
    id SERIAL PRIMARY KEY,
    timestamp TIMESTAMP NOT NULL,
    device_id INT NOT NULL,
    accx FLOAT NOT NULL,
    accy FLOAT NOT NULL,
    accz FLOAT NOT NULL
);
