Get All Users
GET /Users

Get User by Id
GET /Users/<UserId>

Update User by Id
PUT /Users/<UserId>

Delete User by Id
DELETE /Users/<UserId>

Create new User
POST /Users

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL,
    user_infos VARCHAR NOT NULL,
    alarm_sound VARCHAR NOT NULL,
    notification_sound VARCHAR NOT NULL,
    contacts VARCHAR NOT NULL
);
