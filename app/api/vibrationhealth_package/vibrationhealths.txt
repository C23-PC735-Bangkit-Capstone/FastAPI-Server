Get Vibration Health by DeviceId
GET /VibrationHealth/<DeviceId>

Get Vibration Health by DeviceId and Date
GET /VibrationHealth/<DeviceId>/<Date>

Post new Vibration Health
POST /VibrationHealth

CREATE TABLE vibration_health (
    id SERIAL PRIMARY KEY,
    timestamp TIMESTAMPTZ NOT NULL,
    device_id INT NOT NULL,
    health_category VARCHAR NOT NULL,
    health_score INT NOT NULL
);
